def load_json_safe(path: Path, default: Dict[str, Any]) -> Dict[str, Any]:
    try:
        if path.exists():
            return json.loads(path.read_text(encoding="utf-8"))
    except Exception:
        pass
    return default.copy()


def version_tuple(version: str) -> tuple[int, ...]:
    parts = re.findall(r"\d+", str(version or "0"))
    return tuple(int(p) for p in parts) if parts else (0,)


def is_remote_update_available() -> tuple[bool, Dict[str, Any], Dict[str, Any]]:
    local_meta = load_json_safe(LOCAL_VERSION_PATH, {"version": "0.0.0"})
    if not REMOTE_ROOT.exists() or not REMOTE_VERSION_PATH.exists():
        return False, {}, local_meta

    remote_meta = load_json_safe(REMOTE_VERSION_PATH, {"version": "0.0.0"})
    local_version = str(local_meta.get("version", "0.0.0"))
    remote_version = str(remote_meta.get("version", "0.0.0"))
    available = version_tuple(remote_version) > version_tuple(local_version)
    return available, remote_meta, local_meta

class DataFrameModel(QAbstractTableModel):
    def __init__(self, df: Optional[pd.DataFrame] = None):
        super().__init__()
        self._df = df if df is not None else pd.DataFrame()

    def set_dataframe(self, df: Optional[pd.DataFrame]):
        self.beginResetModel()
        self._df = df if df is not None else pd.DataFrame()
        self.endResetModel()

    def rowCount(self, parent=QModelIndex()):
        return 0 if parent.isValid() else len(self._df.index)

    def columnCount(self, parent=QModelIndex()):
        return 0 if parent.isValid() else len(self._df.columns)

    def data(self, index, role=Qt.DisplayRole):
        if not index.isValid() or role not in (Qt.DisplayRole, Qt.EditRole):
            return None
        val = self._df.iat[index.row(), index.column()]
        return "" if pd.isna(val) else str(val)

    def headerData(self, section, orientation, role=Qt.DisplayRole):
        if role != Qt.DisplayRole:
            return None
        if orientation == Qt.Horizontal:
            return str(self._df.columns[section]) if section < len(self._df.columns) else ""
        return str(self._df.index[section])


class WorkerSignals(QObject):
    finished = Signal(object)
    error = Signal(str)


class Worker(QRunnable):
    def __init__(self, fn: Callable, *args, **kwargs):
        super().__init__()
        self.fn = fn
        self.args = args
        self.kwargs = kwargs
        self.signals = WorkerSignals()

    def run(self):
        try:
            result = self.fn(*self.args, **self.kwargs)
            self.signals.finished.emit(result)
        except Exception:
            self.signals.error.emit(traceback.format_exc())

class CollapsibleSection(QWidget):

        self.toggle_btn.clicked.connect(self._on_toggled)
        self.content = content
        self.content.setVisible(expanded)

        layout.setContentsMargins(0, 0, 0, 0)
        layout.setSpacing(4)
        layout.addWidget(self.toggle_btn)
        layout.addWidget(self.content)

    def _on_toggled(self, checked: bool):
        self.toggle_btn.setArrowType(Qt.DownArrow if checked else Qt.RightArrow)
        self.content.setVisible(checked)

class SqlSyntaxHighlighter(QSyntaxHighlighter):
    KEYWORDS = [
        "SELECT", "FROM", "WHERE", "GROUP", "BY", "ORDER", "HAVING", "JOIN", "LEFT", "RIGHT",
        "INNER", "OUTER", "FULL", "ON", "WITH", "AS", "CASE", "WHEN", "THEN", "ELSE", "END",
        "AND", "OR", "NOT", "IN", "EXISTS", "LIKE", "IS", "NULL", "UNION", "ALL", "DISTINCT",
        "INSERT", "INTO", "VALUES", "UPDATE", "SET", "DELETE", "CREATE", "REPLACE", "VIEW",
        "TABLE", "ALTER", "DROP", "MERGE", "USING", "OVER", "PARTITION", "KEEP", "DENSE_RANK",
        "START", "CONNECT", "PRIOR", "LEVEL", "QUALIFY", "LIMIT", "FETCH", "CURRENT_DATE",
        "CURRENT_TIMESTAMP", "SYSDATE", "SYSTIMESTAMP", "NVL", "COALESCE", "DECODE", "IFF"
    ]

    def __init__(self, document):
        super().__init__(document)
        self.rules = []

        kw_format = QTextCharFormat()
        kw_format.setForeground(QColor('#1f4fd6'))
        kw_format.setFontWeight(QFont.Bold)
        for kw in self.KEYWORDS:
            self.rules.append((re.compile(rf'\b{kw}\b', re.IGNORECASE), kw_format))

        fn_format = QTextCharFormat()
        fn_format.setForeground(QColor('#7a3fb0'))
        for patt in [
            re.compile(r'\b[A-Za-z_][A-Za-z0-9_]*\s*(?=\()'),
        ]:
            self.rules.append((patt, fn_format))

        bind_format = QTextCharFormat()
        bind_format.setForeground(QColor('#b54708'))
        bind_format.setFontWeight(QFont.Bold)
        for patt in [
            re.compile(r'(?<!:):[A-Za-z_][A-Za-z0-9_]*'),
            re.compile(r'(?<!&)\&[A-Za-z_][A-Za-z0-9_]*'),
            re.compile(r'\$[A-Za-z_][A-Za-z0-9_]*'),
        ]:
            self.rules.append((patt, bind_format))

        string_format = QTextCharFormat()
        string_format.setForeground(QColor('#0a7f3f'))
        self.rules.append((re.compile(r"'([^']|'')*'"), string_format))

        num_format = QTextCharFormat()
        num_format.setForeground(QColor('#0f766e'))
        self.rules.append((re.compile(r'\b\d+(?:\.\d+)?\b'), num_format))

        comment_format = QTextCharFormat()
        comment_format.setForeground(QColor('#6b7280'))
        self.rules.append((re.compile(r'--[^\n]*'), comment_format))

    def highlightBlock(self, text):
        for pattern, fmt in self.rules:
            for m in pattern.finditer(text):
                self.setFormat(m.start(), m.end() - m.start(), fmt)


class ResultViewerDialog(QDialog):
    def __init__(self, title: str, df: pd.DataFrame, parent: Optional[QWidget] = None):
        super().__init__(parent)
        self.df = df.copy() if df is not None else pd.DataFrame()
        self.setWindowTitle(title)
        self.resize(1200, 760)
        layout = QVBoxLayout(self)

        top = QHBoxLayout()
        self.meta_label = QLabel()
        export_btn = QPushButton("Pobierz Excel")
        export_btn.clicked.connect(self.export_excel)
        top.addWidget(self.meta_label)
        top.addStretch(1)
        top.addWidget(export_btn)
        layout.addLayout(top)

        self.table = QTableView()
        self.model = DataFrameModel(self.df)
        self.table.setModel(self.model)
        self.table.horizontalHeader().setSectionResizeMode(QHeaderView.Interactive)
        self.table.horizontalHeader().setStretchLastSection(False)
        self.table.setAlternatingRowColors(True)
        self.table.setSelectionBehavior(QTableView.SelectItems)
        self.table.setSelectionMode(QTableView.ExtendedSelection)
        layout.addWidget(self.table, 1)

        buttons = QDialogButtonBox(QDialogButtonBox.Close)
        buttons.rejected.connect(self.reject)
        buttons.accepted.connect(self.accept)
        buttons.button(QDialogButtonBox.Close).clicked.connect(self.close)
        layout.addWidget(buttons)
        self._refresh()

    def _refresh(self):
        rows, cols = self.df.shape if self.df is not None else (0, 0)
        self.meta_label.setText(f"Wiersze: {rows}    Kolumny: {cols}")
        self.table.resizeColumnsToContents()

    def export_excel(self):
        path, _ = QFileDialog.getSaveFileName(self, "Zapisz wynik do Excel", "wynik.xlsx", "Excel (*.xlsx)")
        if not path:
            return
        try:
            self.df.to_excel(path, index=False)
            QMessageBox.information(self, "Excel", f"Zapisano plik: {path}")
        except Exception:
            QMessageBox.critical(self, "Excel", traceback.format_exc())

        self.error_box.setReadOnly(True)
        self.error_box.setVisible(False)
        self.error_box.setMinimumHeight(120)

        layout.addWidget(self.error_box)

        self.meta_label.setText(text)
        self.view_btn.setVisible(show_more)

        self._current_df = df.copy() if df is not None else pd.DataFrame()
        self.table_model.set_dataframe(self._current_df)
        self.table.resizeColumnsToContents()
        self.table.resizeRowsToContents()
        self.error_box.clear()
        self.error_box.setVisible(False)
        rows, cols = self._current_df.shape
        self._set_meta(f"Wiersze: {rows} | Kolumny: {cols}", show_more=(rows > 0 or cols > 0))

    def show_error(self, error_text: str):
        self._current_df = pd.DataFrame()
        self.table_model.set_dataframe(pd.DataFrame())
        self.error_box.setVisible(True)
        self.error_box.setPlainText(error_text)
        self._set_meta("Błąd zapytania", show_more=False)

    def clear(self):
        self._current_df = pd.DataFrame()
        self.table_model.set_dataframe(pd.DataFrame())
        self.error_box.clear()
        self.error_box.setVisible(False)
        self._set_meta("Brak danych", show_more=False)

    def _open_viewer(self):
        if self._current_df is None or self._current_df.empty and len(self._current_df.columns) == 0:
            return
        dlg = ResultViewerDialog(self._title, self._current_df, self)
        dlg.exec()

    # -------- connection handling --------
    def connect_oracle(self):
        options = self.config.get("oracle_connections", [])
        combo = QComboBox()
        for item in options:
            combo.addItem(item["name"], item)
        user_edit = QLineEdit()
        pwd_edit = QLineEdit()
        pwd_edit.setEchoMode(QLineEdit.Password)
        dlg = LoginDialog("Połącz z Oracle", [("Połączenie", combo), ("User", user_edit), ("Password", pwd_edit)], self)
        if dlg.exec() != QDialog.Accepted:
            return
        selected = combo.currentData()
        username = user_edit.text().strip()
        password = pwd_edit.text()
        if not selected or not username:
            QMessageBox.warning(self, "Oracle", "Wybierz połączenie i wpisz user.")
            return

        host = selected.get("host", "")
        port = selected.get("port", 1521)
        service_name = selected.get("service_name", "")
        dsn = selected.get("dsn") or oracledb.makedsn(host, int(port), service_name=service_name)
        connect_kwargs = {"user": username, "password": password, "dsn": dsn}
        try:
            conn = oracledb.connect(**connect_kwargs)
            self.oracle_session = OracleSession(selected["name"], username, conn, connect_kwargs=connect_kwargs)
            txt = f"Połączono: {selected['name']} / {username}"
            self.oracle_bar.set_connected(txt)
            self.statusBar().showMessage(txt, 5000)
        except Exception:
            QMessageBox.critical(self, "Oracle", traceback.format_exc())

    def disconnect_oracle(self):
        if self._active_job_kind == "oracle":
            self.cancel_oracle_query()
        if self.oracle_session and self.oracle_session.conn:
            try:
                self.oracle_session.conn.close()
            except Exception:
                pass
        self.oracle_session = None
        self.oracle_bar.set_disconnected()
        self.statusBar().showMessage("Oracle rozłączony", 5000)

    def connect_snowflake(self):
        options = self.config.get("snowflake_connections", [])
        combo = QComboBox()
        for item in options:
            combo.addItem(item["name"], item)

        user_edit = QLineEdit()
        pwd_edit = QLineEdit()
        pwd_edit.setEchoMode(QLineEdit.Password)
        pwd_edit.setPlaceholderText("Hasło lub puste dla externalbrowser")
        auth_combo = QComboBox()
        auth_combo.addItem("Hasło", "password")
        auth_combo.addItem("externalbrowser", "externalbrowser")

        def _sync_auth_widgets():
            use_password = auth_combo.currentData() == "password"
            pwd_edit.setEnabled(use_password)
            if not use_password:
                pwd_edit.clear()

        auth_combo.currentIndexChanged.connect(_sync_auth_widgets)

        dlg = LoginDialog(
            "Połącz ze Snowflake",
            [("Połączenie", combo), ("Login", user_edit), ("Metoda", auth_combo), ("Password", pwd_edit)],
            self,
        )
        selected = combo.currentData()
        if isinstance(selected, dict) and str(selected.get("authenticator") or "").lower() == "externalbrowser":
            auth_combo.setCurrentIndex(1)
            _sync_auth_widgets()
        if dlg.exec() != QDialog.Accepted:
            return

        selected = combo.currentData()
        username = user_edit.text().strip()
        password = pwd_edit.text()
        auth_mode = auth_combo.currentData()
        if not selected or not username:
            QMessageBox.warning(self, "Snowflake", "Wybierz połączenie i wpisz login.")
            return
        if auth_mode == "password" and not password:
            QMessageBox.warning(self, "Snowflake", "Dla logowania hasłem wpisz password albo wybierz externalbrowser.")
            return

        kwargs = {
            "user": username,
            "account": selected.get("account", ""),
            "warehouse": selected.get("warehouse", ""),
            "database": selected.get("database", ""),
            "schema": selected.get("schema", ""),
            "role": selected.get("role") or None,
        }
        if auth_mode == "externalbrowser":
            kwargs["authenticator"] = "externalbrowser"
        else:
            kwargs["password"] = password

        self.snowflake_bar.set_connecting("Łączenie / autoryzacja...")
        self.statusBar().showMessage("Snowflake: otwieram połączenie...", 5000)
        worker = Worker(self._connect_snowflake_worker, selected["name"], username, dict(selected), kwargs)
        self._snowflake_connect_worker = worker
        worker.signals.finished.connect(self._connect_snowflake_success)
        worker.signals.error.connect(self._connect_snowflake_error)
        self.thread_pool.start(worker)

    def _connect_snowflake_worker(self, connection_name: str, username: str, selected: Dict[str, Any], kwargs: Dict[str, Any]):
        conn = SnowflakeDbDriver.connect(**kwargs)
        return {
            "connection_name": connection_name,
            "username": username,
            "selected": selected,
            "kwargs": kwargs,
            "conn": conn,
        }

                pass

            payload["connection_name"],
            payload["username"],
            payload["conn"],
            dict(payload["selected"]),
            connect_kwargs=dict(payload["kwargs"]),
        )
        txt = f"Połączono: {payload['connection_name']} / {payload['username']}"
        self.snowflake_bar.set_connected(txt)
        self.statusBar().showMessage(txt, 5000)

    def _extract_explicit_set_vars(self, sql: str) -> Dict[str, str]:
        values: Dict[str, str] = {}
        for raw_line in (sql or "").splitlines():
            line = raw_line.strip()
            if not line or line.startswith("--"):
                continue
            match = RE_UNCOMMENTED_SET_LINE.match(line)
            if not match:
                continue
            name = match.group(1).upper()
            value_part = re.sub(r"^\s*SET\s+[A-Za-z_][A-Za-z0-9_]*\s*=", "", line, flags=re.IGNORECASE).strip()
            value_part = re.sub(r";\s*$", "", value_part).strip()
            values[name] = value_part
        return values

    def _split_sql_statements(self, sql: str) -> List[str]:
        statements: List[str] = []
        buf: List[str] = []
        in_single = False
        in_double = False
        i = 0
        text = sql or ""
        while i < len(text):
            ch = text[i]
            nxt = text[i + 1] if i + 1 < len(text) else ""
            if not in_double and ch == "'":
                if in_single and nxt == "'":
                    buf.append(ch)
                    buf.append(nxt)
                    i += 2
                    continue
                in_single = not in_single
                buf.append(ch)
            elif not in_single and ch == '"':
                in_double = not in_double
                buf.append(ch)
            elif not in_single and not in_double and ch == '-' and nxt == '-':
                while i < len(text) and text[i] != "\n":
                    buf.append(text[i])
                    i += 1
                continue
            elif not in_single and not in_double and ch == ';':
                stmt = ''.join(buf).strip()
                if stmt:
                    statements.append(stmt)
                buf = []
            else:
                buf.append(ch)
            i += 1
        tail = ''.join(buf).strip()
        if tail:
            statements.append(tail)
        return statements

    # -------- parameter helpers --------
    def _extract_live_params(self, sql: str) -> List[str]:
        seen = []
        explicit_set_vars = set(self._extract_explicit_set_vars(sql).keys())
        for name in extract_oracle_params(sql) or []:
            upper = str(name).upper()
            if upper not in seen:
                seen.append(upper)
        for name in extract_snowflake_variable_names(sql) or []:
            upper = str(name).upper()
            if upper not in explicit_set_vars and upper not in seen:
                seen.append(upper)
        for rx in (RE_ORACLE_AMP_FALLBACK, RE_ORACLE_COLON):
            for m in rx.finditer(sql or ""):
                upper = m.group(1).upper()
                if upper not in seen:
                    seen.append(upper)
        return seen

    def _prompt_for_params(self, sql: str) -> Optional[Dict[str, str]]:
        names = self._extract_live_params(sql)
        if not names:
            return {}
        dlg = ParamsDialog(names, self)
        explicit_set_vars = self._extract_explicit_set_vars(sql)
        for key, value in explicit_set_vars.items():
            widget = dlg._inputs.get(key)
            if widget is not None:
                widget.setText(value)
        if dlg.exec() != QDialog.Accepted:
            return None
        return dlg.values()

    def _current_sf_schema_qualified(self) -> Optional[str]:

    def _normalize_param_values(self, params: Dict[str, str]) -> Dict[str, Any]:
        out: Dict[str, Any] = {}
        for k, v in (params or {}).items():
            out[k] = None if v == "" else v
            out[k.lower()] = None if v == "" else v
            out[k.upper()] = None if v == "" else v
        return out

    def _clean_sql_for_execution(self, sql: str) -> str:
        cleaned = (sql or "").strip()
        cleaned = re.sub(r";\s*$", "", cleaned)
        return cleaned

    def _strip_set_statements(self, sql: str) -> str:
        lines = []
        for line in (sql or "").splitlines():
            if line.strip().upper().startswith("SET "):
                continue
            lines.append(line)
        return "\n".join(lines)

    def _snowflake_literal(self, value: Any) -> str:
        if value is None or value == "":
            return "NULL"
        if isinstance(value, str):
            return "'" + value.replace("'", "''") + "'"
        return str(value)

    def _build_snowflake_replacements(self, params: Dict[str, str]) -> Dict[str, str]:
        replacements: Dict[str, str] = {}
        for name, value in (params or {}).items():
            upper = str(name).upper()
            lower = str(name).lower()
            literal = self._snowflake_literal(None if value == "" else value)
            for token in {
                f"${upper}", f"${lower}", f"${name}",
                f":{upper}", f":{lower}", f":{name}",
                f"&{upper}", f"&{lower}", f"&{name}",
            }:
                replacements[token] = literal
        return dict(sorted(replacements.items(), key=lambda item: len(item[0]), reverse=True))

    def _prepare_snowflake_sql_for_execution(self, conn, sql: str, params: Dict[str, str]) -> str:
        out = sql or ""
        has_oracle_binds = bool(extract_oracle_params(out))
        has_oracle_object_names = bool(ORACLE_REF_REGEX.search(out or ""))
        if has_oracle_binds or has_oracle_object_names:
            out = snf_tr(
                out,
                use_snowflake_vars=False,
                comment_set=False,
                target_database=self._conn_sf_database(conn),
                target_schema=self._conn_sf_schema_qualified(conn),
            )
        explicit_set_vars = self._extract_explicit_set_vars(out)
        out = oracle_params_to_snowflake_vars(out)
        if explicit_set_vars:
            filtered_params = {k: v for k, v in (params or {}).items() if str(k).upper() not in explicit_set_vars}
        else:
            filtered_params = params
        replacements = self._build_snowflake_replacements(filtered_params)
        for token, value in replacements.items():
            out = out.replace(token, value)
        return self._clean_sql_for_execution(out)

    def _fetch_dataframe_from_cursor(self, cur) -> pd.DataFrame:
        if cur.description:
            cols = [d[0] for d in cur.description]
            rows = cur.fetchall()
            return pd.DataFrame(rows, columns=cols)
        return pd.DataFrame([{"STATUS": "OK", "ROWCOUNT": cur.rowcount}])

    def _execute_oracle_sql(self, sql: str, params: Dict[str, str]) -> pd.DataFrame:
        exec_sql = self._clean_sql_for_execution(oracle_to_oracle_bind_sql(sql))
        bind_params = self._normalize_param_values(params)
        conn = oracledb.connect(**self.oracle_session.connect_kwargs)
        self._active_oracle_exec_conn = conn
        try:
            cur = conn.cursor()
            self._active_oracle_exec_cursor = cur
            try:
                cur.execute(exec_sql, bind_params)
                if cur.description:
                    cols = [d[0] for d in cur.description]
                    rows = cur.fetchall()
                    return pd.DataFrame(rows, columns=cols)
                conn.commit()
                return pd.DataFrame([{"STATUS": "OK", "ROWCOUNT": cur.rowcount}])
            finally:
                try:
                    cur.close()
                finally:
                    self._active_oracle_exec_cursor = None
        finally:
            self._active_oracle_exec_conn = None
            conn.close()

    def _execute_snowflake_sql(self, sql: str, params: Dict[str, str]) -> pd.DataFrame:
        conn = SnowflakeDbDriver.connect(**self.snowflake_session.connect_kwargs)
        self._active_snowflake_exec_conn = conn
        self._active_snowflake_query_id = None
        try:
            exec_sql = self._prepare_snowflake_sql_for_execution(conn, sql, params)
            statements = self._split_sql_statements(exec_sql)
            if not statements:
                return pd.DataFrame()

            last_df: Optional[pd.DataFrame] = None
            last_status_df: Optional[pd.DataFrame] = None

            for stmt in statements:
                cur = conn.cursor()
                self._active_snowflake_exec_cursor = cur
                try:
                    cur.execute_async(stmt)
                    self._active_snowflake_query_id = getattr(cur, "sfqid", None)
                    if self._active_snowflake_query_id:
                        while True:
                            status = conn.get_query_status_throw_if_error(self._active_snowflake_query_id)
                            if not conn.is_still_running(status):
                                break
                            time.sleep(0.25)
                        cur.get_results_from_sfqid(self._active_snowflake_query_id)
                    else:
                        cur.execute(stmt)

                    fetched = self._fetch_dataframe_from_cursor(cur)
                    if cur.description:
                        last_df = fetched
                    else:
                        last_status_df = fetched
                finally:
                    try:
                        cur.close()
                    finally:
                        self._active_snowflake_exec_cursor = None
                        self._active_snowflake_query_id = None

            if last_df is not None:
                return last_df
            if last_status_df is not None:
                return last_status_df
            return pd.DataFrame()
        finally:
            self._active_snowflake_exec_conn = None
            conn.close()

    def cancel_oracle_query(self):
        conn = self._active_oracle_exec_conn
        if conn is None:
            self.statusBar().showMessage("Oracle: brak aktywnego zapytania", 3000)
            return
        self._oracle_cancel_requested = True
        try:
            conn.cancel()
            self.statusBar().showMessage("Oracle: przerwanie zapytania...", 5000)
        except Exception:
            try:
                conn.close()
                self.statusBar().showMessage("Oracle: zamykam połączenie do przerwania zapytania...", 5000)
            except Exception:
                self.statusBar().showMessage("Oracle: nie udało się przerwać zapytania", 5000)

    def cancel_snowflake_query(self):
        conn = self._active_snowflake_exec_conn
        query_id = self._active_snowflake_query_id
        if conn is None:
            self.statusBar().showMessage("Snowflake: brak aktywnego zapytania", 3000)
            return
        self._snowflake_cancel_requested = True
        cancelled = False
        if query_id:
            sql = f"SELECT SYSTEM$CANCEL_QUERY('{query_id}')"
            try:
                with conn.cursor() as cancel_cur:
                    cancel_cur.execute(sql)
                cancelled = True
            except Exception:
                pass
            if not cancelled:
                try:
                    cancel_conn = SnowflakeDbDriver.connect(**self.snowflake_session.connect_kwargs)
                    try:
                        with cancel_conn.cursor() as cancel_cur:
                            cancel_cur.execute(sql)
                        cancelled = True
                    finally:
                        cancel_conn.close()
                except Exception:
                    cancelled = False
        if not cancelled:
            try:
                conn.close()
                cancelled = True
            except Exception:
                cancelled = False
        if cancelled:
            self.statusBar().showMessage("Snowflake: przerwanie zapytania...", 5000)
        else:
            self.statusBar().showMessage("Snowflake: nie udało się przerwać zapytania", 5000)

    def _run_db_job(self, fn: Callable[[], pd.DataFrame], results_pane: ResultsPane, label: str, kind: str, scope: str):
        self.statusBar().showMessage(f"Uruchamianie: {label}...")
        results_pane.clear()
        self._set_busy(True, f"{label}: wykonywanie zapytania...", scope=scope, kind=kind)
        if kind == "oracle":
            self._oracle_cancel_requested = False
        if kind == "snowflake":
            self._snowflake_cancel_requested = False
        worker = Worker(fn)
        self._active_worker = worker
        worker.signals.finished.connect(lambda df: self._job_success(df, results_pane, label))
        worker.signals.error.connect(lambda err: self._job_error(err, results_pane, label))
        self.thread_pool.start(worker)

    def _is_cancelled_job(self, label: str, err: str) -> bool:
        text = (err or "").upper()
        if label.lower() == "oracle":
            return self._oracle_cancel_requested or "ORA-01013" in text or "USER REQUESTED CANCEL" in text
        return self._snowflake_cancel_requested or "CANCEL" in text or "ERRNO = 604" in text or "QUERY WAS CANCELED" in text

    def _job_success(self, df: pd.DataFrame, results_pane: ResultsPane, label: str):
        self._set_busy(False)
        self._active_worker = None
        self._oracle_cancel_requested = False
        self._snowflake_cancel_requested = False
        results_pane.show_dataframe(df)
        self.statusBar().showMessage(f"{label}: gotowe ({len(df)} wierszy)", 5000)

    def _job_error(self, err: str, results_pane: ResultsPane, label: str):
        self._set_busy(False)
        self._active_worker = None
        cancelled = self._is_cancelled_job(label, err)
        self._oracle_cancel_requested = False
        self._snowflake_cancel_requested = False
        if cancelled:
            results_pane.show_error("Zapytanie przerwane przez użytkownika.")
            self.statusBar().showMessage(f"{label}: przerwano", 5000)
        else:
            results_pane.show_error(err)
            self.statusBar().showMessage(f"{label}: błąd", 5000)

    def scan_folder(self):
        base = Path(self.folder_path.text().strip())
        if not base.exists() or not base.is_dir():
            QMessageBox.warning(self, "Folder", "To nie jest poprawny katalog.")
            return
        self.file_list.clear()
        pattern = "**/*.sql" if self.folder_recursive.isChecked() else "*.sql"
        found = 0
        for p in base.glob(pattern):
            if not p.is_file():
                continue
            txt = p.read_text(encoding="utf-8", errors="ignore")
            refs = sorted(set(ORACLE_REF_REGEX.findall(txt)))
            params = self._extract_live_params(txt)
            if refs:
                item = QListWidgetItem(f"{p.name} (refs:{len(refs)}) — {p}")
                item.setData(Qt.UserRole, {"path": str(p), "refs": refs, "params": params})
                self.file_list.addItem(item)
                found += 1
        self.statusBar().showMessage(f"Znalezione pliki: {found}", 5000)
    
    def load_selected_file(self, current: Optional[QListWidgetItem], previous: Optional[QListWidgetItem] = None):
        if not current:

        data = current.data(Qt.UserRole) or {}
        path = Path(data.get("path", ""))

            return
        txt = path.read_text(encoding="utf-8", errors="ignore")
        self.oracle_editor_2.editor.setPlainText(txt)
        self.snowflake_editor_2.editor.clear()
        refs = data.get("refs", [])
        params = data.get("params", [])
        parts = []
        if refs:
            parts.append("Refs:\n" + "\n".join(refs[:200]))
        if params:
            parts.append("\nParametry:\n" + "\n".join(params))
        self.file_refs.setPlainText("\n\n".join(parts))
        self.oracle_results_2.clear()
        self.snowflake_results_2.clear()

    def perform_application_update(self, remote_meta: Optional[Dict[str, Any]] = None):
        remote_meta = remote_meta or load_json_safe(REMOTE_VERSION_PATH, {"version": "0.0.0"})
        try:
            if not LOCAL_UPDATE_EXE.exists():
                QMessageBox.warning(
                    self,
                    "Aktualizacja",
                    f"Nie znaleziono pliku update.exe w katalogu aplikacji:\n{LOCAL_UPDATE_EXE}"
                )
                return

            version_text = str(remote_meta.get("version", "0.0.0"))
            creationflags = 0
            if os.name == "nt":
                creationflags = getattr(subprocess, "CREATE_NO_WINDOW", 0)

            subprocess.Popen([str(LOCAL_UPDATE_EXE)], cwd=str(APP_DIR), creationflags=creationflags)
            QApplication.instance().quit()
        except Exception:
            QMessageBox.critical(self, "Błąd aktualizacji", traceback.format_exc())
