        function disableButtons() {
            // Znajdź wszystkie przyciski na stronie
            const buttons = document.querySelectorAll("button");

            // Przeiteruj przez wszystkie przyciski
            buttons.forEach(button => {
                // Sprawdź, czy przycisk ma klasę 'btn-primary'
                if (button.classList.contains('btn-primary')) {
                    // Zmień klasę na 'btn btn-primary disabled'
                    button.classList.add('disabled');

                    // Zmieniamy tekst na "Loading..." dla przycisku submit
                    if (button.type === "submit") {
                        button.innerText = "Loading...";
                    }
                }
            });
        }
