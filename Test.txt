import webview

# Funkcja JavaScript do obsługi Ctrl + Backspace
js_code = """
document.addEventListener('keydown', function(event) {
    if (event.ctrlKey && event.key === 'Backspace') {
        window.history.back();
    }
});
"""

def on_loaded():
    # Inicjalizacja funkcji JavaScript po załadowaniu okna
    webview.windows[0].evaluate_js(js_code)

if __name__ == '__main__':
    # Uruchomienie okna pywebview
    window = webview.create_window('PyWebView Example', 'https://example.com', width=800, height=600)
    webview.start(on_loaded)
