import webview

def copy_text():
    # JavaScript function to copy text
    js_code = """
    function copyText() {
        var text = window.getSelection().toString(); // Get selected text
        var input = document.createElement('input');
        document.body.appendChild(input);
        input.setAttribute('value', text);
        input.select();
        document.execCommand('copy');
        document.body.removeChild(input);
    }
    copyText();
    """
    # Execute JavaScript code in the webview
    window.evaluate(js_code)

# Create and show the webview
window = webview.create_window('My WebView', 'https://example.com')
webview.start(copy_text)


import webview

def zoom_in():
    # JavaScript function to zoom in
    js_code = """
    document.body.style.zoom = parseFloat(getComputedStyle(document.body).zoom || 1) + 0.1;
    """
    window.evaluate(js_code)

def zoom_out():
    # JavaScript function to zoom out
    js_code = """
    document.body.style.zoom = parseFloat(getComputedStyle(document.body).zoom || 1) - 0.1;
    """
    window.evaluate(js_code)

def setup_shortcuts():
    # JavaScript to listen for Ctrl+ and Ctrl- shortcuts
    js_code = """
    document.addEventListener('keydown', function(e) {
        if (e.ctrlKey && e.key === '+') {
            document.body.style.zoom = parseFloat(getComputedStyle(document.body).zoom || 1) + 0.1;
            e.preventDefault();
        } else if (e.ctrlKey && e.key === '-') {
            document.body.style.zoom = parseFloat(getComputedStyle(document.body).zoom || 1) - 0.1;
            e.preventDefault();
        }
    });
    """
    window.evaluate(js_code)

def load_page():
    # Load the page and set up shortcuts
    setup_shortcuts()

# Create and show the webview
window = webview.create_window('My WebView', 'https://example.com')
webview.start(load_page)
